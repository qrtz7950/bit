//방법 3 (EcmaScript6 추가)
	function func(num1 = 0, num2 = 0, num3 = 0){
		alert(`num1 : ${num1}, num2: ${num2}, num3: ${num3}`);
	}

	위의 func(num1...)에서 num1=0 으로 쓰면 2,3도 =0을 붙여서 초기화 해줘야하고
	num2=0만 붙인경우 오른쪽에 있는 num3에도 =0을 붙여 초기화 해줘야한다.

	1. 타이머
		setTimeout(function , ms) //ms만큼 지나면 function을 수행한다.
		setInterval(function, ms) //ms마다 function을 수행한다.
	2. 코드 실행함수
		eval(string) //string에 내용이 js의 문법이라면 그 문법을 실행한다
	3. 숫자변환함수
		parseInt(string)
